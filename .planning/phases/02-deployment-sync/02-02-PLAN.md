    ```python
        """Get version history for a document (newest first)."""
        async with self.session() as session:
    ```

  <action>
    Create a background sync manager that runs periodically:

    ```python
    import asyncio
    from datetime import datetime, timezone
    import structlog
    from typing import Callable

    logger = structlog.get_logger(__name__)

    class BackgroundSyncManager:
        """Manages background synchronization of documents from configured sources."""
        
        def __init__(
            self,
            sync_interval_seconds: int = 3600,  # 1 hour default
            sources: list[str] | None = None,  # ['confluence', 'jira', 'notion']
        ):
            self.sync_interval = sync_interval_seconds
            self.sources = sources or ['confluence', 'jira', 'notion']
            self._task: asyncio.Task | None = None
            self._running = False
        
        async def start(self) -> None:
            """Start the background sync task."""
            if self._running:
                logger.warning("BackgroundSyncManager already running")
                return
            
            self._running = True
            self._task = asyncio.create_task(self._sync_loop())
            logger.info("BackgroundSyncManager started", interval=self.sync_interval)
        
        async def stop(self) -> None:
            """Stop the background sync task."""
            if not self._running:
                return
            
            self._running = False
            if self._task:
                self._task.cancel()
                try:
                    await self._task
                except asyncio.CancelledError:
                    pass
            logger.info("BackgroundSyncManager stopped")
        
        async def _sync_loop(self) -> None:
            """Run sync loop indefinitely."""
            while self._running:
                try:
                    await asyncio.sleep(self.sync_interval)
                    if self._running:
                        await self.sync_all_sources()
                except asyncio.CancelledError:
                    break
                except Exception as exc:
                    logger.error("Sync loop error", error=str(exc))
        
        async def sync_all_sources(self) -> None:
            """Sync all configured sources."""
            logger.info("Starting scheduled sync", sources=self.sources)
            
            for source in self.sources:
                try:
                    await self._sync_source(source)
                except Exception as exc:
                    logger.error("Sync failed", source=source, error=str(exc))
        
        async def _sync_source(self, source: str) -> None:
            """Sync a single source (delegate to existing sync logic)."""
            # Import dynamically to avoid circular imports
            from metatron.connectors.registry import CONNECTORS
            
            if source not in CONNECTORS:
                logger.warning("Unknown sync source", source=source)
                return
            
            connector = CONNECTORS[source]
            await connector.sync()  # Assuming sync() method exists
            logger.info("Sync completed", source=source)
    ```

    Modify existing sync logic to capture document versions when documents change:

    When syncing documents from any source (Confluence, Jira, Notion), the sync pipeline should:
    1. Fetch document from source
    2. Compare content hash with latest stored version
    3. If content changed: call `postgres.store_document_version()` with `sync_source='confluence'` (or jira/notion)
    4. Update document in Qdrant/Memgraph/PostgreSQL as usual

    Example pattern in existing sync code:
    ```python
    async def sync_source(connector, source_name):
        docs = await connector.fetch()
        for doc in docs:
            # Calculate hash
            new_hash = hashlib.sha256(doc.content.encode()).hexdigest()
            
            # Get latest version
            latest = await postgres.get_latest_version(doc.id)
            
            # If changed, store version
            if not latest or latest.content_hash != new_hash:
                await postgres.store_document_version(
                    document_id=doc.id,
                    content=doc.content,
                    changed_fields={'content': (latest.content if latest else '', doc.content)},
                    sync_source=source_name,
                )
            
            # Update document normally
            await update_document(doc)
    ```
