    ```python
    async def store_document_version(
        self,
        document_id: UUID,
        content: str,
        changed_fields: dict[str, Any] | None = None,
        sync_source: str = "manual",
    ) -> DocumentVersion:
        """Store a new version of a document."""
        import hashlib
        from datetime import datetime, timezone
        import structlog
        
        logger = structlog.get_logger(__name__)
        
        # Calculate content hash
        content_hash = hashlib.sha256(content.encode()).hexdigest()
        
        async with self.session() as session:
            # Get next version number
            result = await session.execute(
                text("""
                    SELECT COALESCE(MAX(version_number), 0) as max_version
                    FROM document_versions
                    WHERE document_id = :doc_id
                """),
                {"doc_id": str(document_id)}
            )
            max_version = result.scalar() or 0
            version_number = max_version + 1
            
            # Create version record
            now = datetime.now(timezone.utc)
            
            result = await session.execute(
                text("""
                    INSERT INTO document_versions 
                    (id, document_id, version_number, content, content_hash, created_at, changed_fields, sync_source)
                    VALUES (:id, :doc_id, :version_num, :content, :hash, :created, :fields, :source)
                    RETURNING id
                """),
                {
                    "id": str(uuid4()),
                    "doc_id": str(document_id),
                    "version_num": version_number,
                    "content": content,
                    "hash": content_hash,
                    "created": now,
                    "fields": json.dumps(changed_fields or {}),
                    "source": sync_source,
                }
            )
            
            await session.commit()
            version_id = result.scalar()
            
            logger.info(
                "Document version stored",
                document_id=str(document_id),
                version_number=version_number,
                sync_source=sync_source,
            )
            
            return DocumentVersion(
                id=UUID(version_id),
                document_id=document_id,
                version_number=version_number,
                content=content,
                content_hash=content_hash,
                created_at=now,
                changed_fields=changed_fields or {},
                sync_source=sync_source,
            )
    ```

    ```python
    async def get_document_history(
        self,
        document_id: UUID,
        limit: int = 10,
        offset: int = 0,
    ) -> tuple[list[DocumentVersion], int]:
        """Get version history for a document (newest first)."""
        import structlog
        from datetime import timezone
        
        logger = structlog.get_logger(__name__)
        
        async with self.session() as session:
            # Get total count
            count_result = await session.execute(
                text("""
                    SELECT COUNT(*) as total
                    FROM document_versions
                    WHERE document_id = :doc_id
                """),
                {"doc_id": str(document_id)}
            )
            total = count_result.scalar() or 0
            
            # Get paginated versions (newest first)
            result = await session.execute(
                text("""
                    SELECT id, document_id, version_number, content, content_hash, 
                           created_at, changed_fields, sync_source
                    FROM document_versions
                    WHERE document_id = :doc_id
                    ORDER BY version_number DESC
                    LIMIT :limit OFFSET :offset
                """),
                {
                    "doc_id": str(document_id),
                    "limit": limit,
                    "offset": offset,
                }
            )
            
            versions = []
            for row in result:
                versions.append(DocumentVersion(
                    id=UUID(row[0]),
                    document_id=UUID(row[1]),
                    version_number=row[2],
                    content=row[3],
                    content_hash=row[4],
                    created_at=row[5].replace(tzinfo=timezone.utc) if row[5] else None,
                    changed_fields=json.loads(row[6]) if row[6] else {},
                    sync_source=row[7],
                ))
            
            logger.info(
                "Document history retrieved",
                document_id=str(document_id),
                count=len(versions),
                total=total,
            )
            
            return versions, total
    ```

    ```python
    async def get_latest_version(self, document_id: UUID) -> DocumentVersion | None:
        """Get the latest version of a document."""
        import structlog
        from datetime import timezone
        
        logger = structlog.get_logger(__name__)
        
        async with self.session() as session:
            result = await session.execute(
                text("""
                    SELECT id, document_id, version_number, content, content_hash,
                           created_at, changed_fields, sync_source
                    FROM document_versions
                    WHERE document_id = :doc_id
                    ORDER BY version_number DESC
                    LIMIT 1
                """),
                {"doc_id": str(document_id)}
            )
            
            row = result.first()
            if not row:
                logger.debug("No versions found for document", document_id=str(document_id))
                return None
            
            version = DocumentVersion(
                id=UUID(row[0]),
                document_id=UUID(row[1]),
                version_number=row[2],
                content=row[3],
                content_hash=row[4],
                created_at=row[5].replace(tzinfo=timezone.utc) if row[5] else None,
                changed_fields=json.loads(row[6]) if row[6] else {},
                sync_source=row[7],
            )
            
            logger.debug(
                "Latest version retrieved",
                document_id=str(document_id),
                version_number=version.version_number,
            )
            
            return version
    ```

    ```python
    from fastapi import APIRouter, HTTPException, Query
    from uuid import UUID
    import structlog
    from metatron.storage.postgres import PostgreSQLStorage
    
    router = APIRouter(prefix="/api/v1", tags=["documents"])
    logger = structlog.get_logger(__name__)
    
    @router.get("/documents/{document_id}/history")
    async def get_document_history(
        document_id: UUID,
        limit: int = Query(10, ge=1, le=100),
        offset: int = Query(0, ge=0),
        postgres: PostgreSQLStorage = Depends(get_postgres_storage),
    ):
        """Get version history for a document.
        
        Returns paginated list of document versions (newest first).
        """
        try:
            versions, total = await postgres.get_document_history(
                document_id=document_id,
                limit=limit,
                offset=offset,
            )
            
            logger.info(
                "Document history endpoint called",
                document_id=str(document_id),
                version_count=len(versions),
            )
            
            return {
                "document_id": str(document_id),
                "versions": [
                    {
                        "version_number": v.version_number,
                        "created_at": v.created_at.isoformat() if v.created_at else None,
                        "sync_source": v.sync_source,
                        "changed_fields": v.changed_fields,
                        "content_preview": v.content[:200] + "..." if len(v.content) > 200 else v.content,
                    }
                    for v in versions
                ],
                "total": total,
                "offset": offset,
                "limit": limit,
                "has_more": (offset + limit) < total,
            }
        except Exception as exc:
            logger.error("Failed to get document history", error=str(exc), document_id=str(document_id))
            raise HTTPException(status_code=500, detail="Failed to retrieve document history")
    ```
